python dna_baselines.py --preset hyenadna-small-32k --window_bps 512,1024,2048 --add_variant_features --classifier mlp --mlp_hidden 256,128 --standardize --no_class_weight --report_auroc --report_auprc

python dna_baselines.py --preset dnabert2 --window_bps 512 --add_variant_features --add_pll_delta --pll_max_positions 16 --report_auroc --report_auprc
